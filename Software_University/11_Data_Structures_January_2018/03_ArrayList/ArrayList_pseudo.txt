class ArrayList {
	int DEF_CAPACITY
	T[] elements
	int size

	ArrayList() { elements = T[DEF_CAPACITY]}
	ArrayList(int capacity)

	add(T ele) {
		if size == elements.size
			grow()
		
		elements[size] = ele
		size+=
	}

	get(int index){
		if (index <= 0 || size >= index)
			throw
		return (T) elements[index]
	}

	grow() {
		int newSize = size * 2
		elements = Arrays.copy(elements, newSize)
	}
}
